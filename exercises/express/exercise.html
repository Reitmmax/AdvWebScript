HTTP Server

m08_app.js

		
// JavaScript Document
'use strict'
const express = require('express');
const app = express();

//shorthand for code above
//const app = require('express')();
const path = require('path');

//allows all files to be read and used
app.use(express.static('./indexbanner'));

app.listen(8080, () => {
    console.log('server listening at 8080');
});

app.get('/',(req, res)=>{
    //res.send('<h1>Welcome to home page</h1>');
	// res.status(200).send('<h1>Welcome to home page</h1>');
	res.status(200).sendFile(path.resolve(__dirname, './indexbanner/index.html'));
	
})
app.get('/about',(req, res)=>{
    //res.send('<h1>Welcome to home page</h1>');
	 //res.status(200).send('<h1>About Me!</h1>');
	res.status(200).sendFile(path.resolve(__dirname, './indexbanner/about.html'));
})

app.all('*',(req,res)=>{
    //res.status(404).send('<h1>Resource Not Found</h1><p>Sorry about that. Do you want to go <a href="/">Home</a>?');
	res.status(200).sendFile(path.resolve(__dirname, './indexbanner/404.html'));
})

This is the code for my server! We are using express this time! The code allows you to go to the page that has the same slug as the functions. If you use "/about" it will take you to the about page, if you use "/", you will go to the home page, and if you use anything else you will see an error page. Compared to my last project using an HTTP server to display the pages, you will see that there is siginically less code. Thanks to Express, it allowed to to simplify the code more! I do not need to call every asset anymore because it does it for me as long as it has the folder and/or file.
